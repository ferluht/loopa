28bool read_xy_file(const std::string & filename, std::vector<int> & tab_sizes,
29 std::vector<double> & tab_mflops, bool quiet = false)
30{
31
32 std::ifstream input_file (filename.c_str(),std::ios::in);
33
34 if (!input_file){
35 if (!quiet) {
36 INFOS("!!! Error opening "<<filename);
37 }
38 return false;
39 }
40
41 int nb_point=0;
42 int size=0;
43 double mflops=0;
44
45 while (input_file >> size >> mflops ){
46 nb_point++;
47 tab_sizes.push_back(size);
48 tab_mflops.push_back(mflops);
49 }
50 SCRUTE(nb_point);
51
52 input_file.close();
53 return true;
54}
