349 auto x = imap.find(cmd.status);
350 if (x == imap.end()) return MIDISTATUS::DONE;

352 if (xy == x->second.end()) return MIDISTATUS::DONE;
353 HardwareControl * ctrl = xy->second;
354 if (ctrl) {
355 ctrl->midiIn(cmd);
356 for (auto it = ctrl->ctrl_seqs.begin(); it < ctrl->ctrl_seqs.end(); it ++) {
357 bool flag = true;
358 for (auto xy = it->first.begin(); xy < it->first.end(); xy ++) {
359 HardwareControl * c = imap.find(xy->first)->second.find(xy->second)->second;
360 if (!c->isPressed())
361 flag = false;
362 }
