326 MIDISTATUS midiIn(MData& cmd) override {
327 auto x = imap.find(cmd.status);
328 if (x == imap.end()) return MIDISTATUS::DONE;
329 auto xy = x->second.find(cmd.data1);
330 if (xy == x->second.end()) return MIDISTATUS::DONE;
331 HardwareControl * ctrl = xy->second;
332 if (ctrl) {
333 ctrl->midiIn(cmd);
334 for (auto it = ctrl->ctrl_seqs.begin(); it < ctrl->ctrl_seqs.end(); it ++) {
335 bool flag = true;
336 for (auto xy = it->first.begin(); xy < it->first.end(); xy ++) {
337 HardwareControl * c = imap.find(xy->first)->second.find(xy->second)->second;
338 if (!c->isPressed())
339 flag = false;
340 }
341 if (flag) {
342 cmd.status = it->second.first;
343 cmd.data1 = it->second.second;
344 break;
345 }
346 }
