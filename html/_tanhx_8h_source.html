38 inline void fplowshelf(float gain) {
39 float sgain = std::sqrt(gain);
40 A = 1;

46 inline void performBLT(float freq_k) {
47 float w0 = 2 * M_PI * freq_k;
48 float KK = (w0 < M_PI) ? (1 / std::tan(w0 / 2)) : 0;
49 a0 = B + A * KK;
50 a1 = B - A * KK;
51 b0 = D + C * KK;
52 b1 = D - C * KK;
53
54 if (b0 != 0) {
55 a0 = a0 / b0;
56 a1 = a1 / b0;
57 b1 = -b1 / b0;
58 } else {
59 a0 = 1;
60 a1 = 0;
61 b1 = 0;
62 }
63 }

65 inline void lowshelf(float freq_k, float gain) {
66 fplowshelf(gain);
67 performBLT(freq_k);
68 }

70 inline float getAmp(float freq_k) {
71 float w = 2 * M_PI * freq_k;
72 float cw = cos(w);
73 return (a0 * a0 + 2 * a0 * a1 * cw + a1 * a1) / (1 - 2 * b1 * cw + b1 * b1);
74 }
